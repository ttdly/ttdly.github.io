对于一个序列 {1, 2, 3, 4, 5} 生成它的全排列，这里只列出三个方法：自底向上生成排列（插入）、Johnson-Trotter 算法、字典序算法。

## 自底向上生成排列
```c++
#include <vector>
#include <iostream>
using namespace std;
vector<vector<int>> result;
vector<int>::iterator it;

void AllSort(int n) {
  vector<int> curr;
  for (int i = 1; i < n; i++) {
    vector<vector<int>> temp;
    int j = 0;
    for (int k = 0; k < result.size(); k++) {
      for (int j = 0; j < i + 1; j++) {
        curr = result[k];
        it = curr.begin() + j;
        curr.insert(it, i + 1);
        temp.push_back(curr);
      }
    }
    result = temp;
  }
}

int main() {
  result.push_back({1});
  AllSort(4);
  for (int n = 0; n < result.size(); n++) {
    cout << '{';
    for (int m = 0; m < result[n].size(); m++) {
      cout << result[n][m];
      if (m != result[n].size() - 1)
        cout << ',';
    }
    cout << '}' << endl;
  }
  return 0;
}
```

## Johnson-Trotter 算法
```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void swap(int& a, int& b) {
  int temp = a;
  a = b;
  b = temp;
}

bool IsMovable(int arr[], int mark[], int index, int n) {
  int next = index + mark[index];
  if (next < 0) return false;
  if (next > n - 1) return false;
  return arr[index] > arr[next];
}

void RevertMark(int &mark) {
  if (mark == -1) mark = 1;
  else mark = -1;
}

void PrintArr(int arr[], int n) {
  cout << '{';
  for (int i = 0; i < n; i++) {
    cout << arr[i];
    if (i != n - 1) cout << ',';
  }
  cout << '}' << endl;
}

void JT(int n) {
  int* arr = (int*)malloc(sizeof(int) * n);
  int* mark = (int*)malloc(sizeof(int) * n);
  int count = n;
  for (int i = 0; i < n; i++) {
    arr[i] = i + 1;
    mark[i] = -1;
  }
  PrintArr(arr, n);
  while(count>0){
    int k = -1, index = -1;
    for (int i = 0; i < n; i++) {
      if (IsMovable(arr, mark, i, n) && arr[i] > k) {
        k = arr[i], index = i;
      }
    }
    int next = index + mark[index];
    while (IsMovable(arr, mark, index, n)) {
      swap(arr[index], arr[next]);
      swap(mark[index], mark[next]);
      index = next;
      next = index + mark[index];
      count--;
      PrintArr(arr, n);
      //PrintArr(mark, n);
    }
    for (int i = 0; i < n; i++) {
      if (arr[i] > k) RevertMark(mark[i]);
      if (IsMovable(arr, mark, i, n)) count++;
    }
  }

}
int main() {
  JT(3);
  return 0;
}
```

## 字典序算法
```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void swap(int& a, int& b) {
  int temp = a;
  a = b;
  b = temp;
}

bool ContinueSort(int arr[], int n) {
  for (int i = 0; i < n-1; i++) {
    if (arr[i] < arr[i + 1]) return true;
  }
  return false;
}

void PrintArr(int arr[], int n) {
  cout << '{';
  for (int i = 0; i < n; i++) {
    cout << arr[i];
    if (i != n - 1) cout << ',';
  }
  cout << '}' << endl;
}

void ZiDianSort(int arr[], int n) {
  int a = 0, b = 0;
  while(ContinueSort(arr,n)){
    for (int i = n - 1; i > 0; i--) {
      if (arr[i] > arr[i - 1]) {
        a = i - 1;
        break;
      }
    }
    for (int j = n - 1; j > 0; j--) {
      if (arr[j] > arr[a]) {
        b = j;
        break;
      }
    }
    swap(arr[a], arr[b]);
    a++;
    b = n-1;
    while (a < b) {
      swap(arr[a++], arr[b--]);
    }
    PrintArr(arr, n);
  }
}

int main() {
  int n = 4;
  int* arr = (int*)malloc(sizeof(int) * n);
  for (int i = 0; i < n; i++) {
    arr[i] = i + 1;
  }
  PrintArr(arr, n);
  ZiDianSort(arr,n);
  return 0;
}
```
